CanSimplify[a_, b_] :=
  10 <= b <= 99 && 10 <= a < b &&
  Block[{a1 = Floor[a/10], a0 = Mod[a, 10], b1 = Floor[b/10], b0 = Mod[b, 10]},
    a0 == b0 && b0 != 0 && b1 != 0 && a1/b1 == a/b ||
    a0 == b1 && b0 != 0 && a1/b0 == a/b ||
    a1 == b0 && b1 != 0 && a0/b1 == a/b ||
    a1 == b1 && b0 != 0 && a0/b0 == a/b]
