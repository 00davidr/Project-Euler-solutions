EightyNineQ[1] := False
EightyNineQ[89] := True
EightyNineQ[x_] := EightyNineQ[Total[Map[(#^2) &, IntegerDigits[x]]]]
Length[Select[Range[10^7 - 1], EightyNineQ]]
