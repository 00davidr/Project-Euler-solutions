WinProb[n_, i_, b_] = If[i == n, Boole[2b > n],
  (i + 1) / (i + 2) * WinProb[n, i + 1, b + 0] +
  1       / (i + 2) * WinProb[n, i + 1, b + 1]];
Floor[1/WinProb[15, 0, 0]]
